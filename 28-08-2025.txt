Q. Counting Spaces
package Strings;
import java.util.*;
public class Trimmimg {
    public static void main(String[] args) {
        String str = "  Hello Java   ";
        int stSpace = 0;
        int endSpace = 0;
        String str2="";
        for(int i=2;i<str.length()-2;i++){
            str2 = str2+str.charAt(i);
        }
        for(int i=0;i<str.length();i++){
            if(str.charAt(i)==' '){
                stSpace++;
            }else{
                break;
            }

        }
        System.out.println("Starting Space: "+stSpace);
        System.out.println("+++++++++++");
        for(int i=str.length()-1;i>0;i--){
            if(str.charAt(i)==' '){
                endSpace++;
            }else{
                break;
            }
        }
        System.out.println("Ending Space: "+endSpace);
        System.out.println("++++++++++");
        System.out.println("The actual String:"+str);
        System.out.println("The Trimmed String:"+str2);
    }
}

Ending Space: 3
++++++++++

Q. Remove all special chars from a given String
package Strings;
import java.util.*;
public class RemoveSpecialCharacters {
    public static void main(String[] args) {
        String str = "Yash12 !@#$%^&*() 1565 yash 2111";
        String str2 = "";
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
            if((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')||(ch >= '0' && ch <= '9')){
                str2=str2+ch;
            }
        }
        System.out.println("The String without special Characters:" +str2);
    }
}

The String without special Characters:Yash121565yash2111
